  • if else if
    •       if(조건){
        명령문  // 조건이 참일 경우
      } else if(조건 2) {
        명령문  // 조건 2일 경우
      } else if(조건 3) {
        명령문  // 조건 3일 경우
      } else {
        명령문  // 거짓일 경우
      }
